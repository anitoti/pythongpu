# animate_coupling.py writes video through cv2.VideoWriter. Two deliberate
# deviations from the dev box's opencv-python==4.5.2.54:
#   - headless build: same cv2 API and VideoWriter, minus the GUI/X11 link that
#     has nothing to resolve against on a batch node.
#   - 4.5.4.60, not 4.5.2.54: 4.5.2.54 ships no cp310 wheel (cp36-cp39 only),
#     so on ACRES's Python 3.10 pip would fall back to compiling OpenCV from
#     source and fail. 4.5.4.60 is the earliest release with a cp310 wheel, so
#     it is the smallest step off the dev box that installs at all.
opencv-python-headless==4.5.4.60

# The dev box still has pytest 4.6.9 (the distro package). That is old enough
# to be worth leaving behind rather than reproducing — nothing in tests/ needs
# it, and 4.6 predates fixtures these tests would otherwise be free to use.
pytest==7.4.3
