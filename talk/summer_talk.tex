\begin{frame}{The Science: Stability and Nimbleness at Once}
  \GeoPanels
  \vspace{0.25cm}
  Your brain holds a thought steadily --- and drops it the instant something
  matters. Those demands fight each other.
  \vspace{0.3cm}
  \begin{columns}[T]
    \column{0.47\textwidth}
      \begin{block}{Too stable}
        Can't leave without a big push. Robust, but you'd never notice the fire alarm.
      \end{block}
    \column{0.47\textwidth}
      \begin{block}{Too unstable}
        Noise alone destroys it. Responsive, but you couldn't hold a thought.
      \end{block}
  \end{columns}
  \vspace{0.3cm}
  The brain does both, in milliseconds, with \textbf{no central controller}. So
  the resolution must be built into the \emph{geometry} of the system.
  \vfill
  {\footnotesize\color{Rust}\textit{``Small changes in current state caused by environmental
  stimuli would be enough to switch between distinct stable brain states.''}\\
  \hfill --- Bollt, Fish, Kumar, Roque dos Santos \& Laurienti,
  \textit{Fractal Basins as a Mechanism for the Nimble Brain} (arXiv:2311.00061)}
\end{frame}
\note{Hook: stability and sensitivity are opposites and the brain refuses to choose.
No homunculus flips the switch. Name the paper here — it's my advisor's, and my

\begin{frame}{Measuring Crinkliness --- and an Ambiguity to Remember}
  \GeoPanels
  \vspace{0.2cm}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{Box-counting dimension.} Cover the boundary with boxes of side
      $\varepsilon$; $N(\varepsilon)$ = boxes touched. If
      $N(\varepsilon)\sim\varepsilon^{-D}$,
      \[ D_0=\lim_{\varepsilon\to0}\frac{\log N(\varepsilon)}{\log(1/\varepsilon)} \]
      Smooth curve $\Rightarrow D=1$. Area-filling $\Rightarrow D=2$. Fractal in between.
    \column{0.44\textwidth}
      \begin{block}{\alert{Hold onto this}}
        $D\to2$ has \textbf{two} explanations:
        \smallskip
        \begin{enumerate}
          \item an intricately interwoven boundary --- the exciting one;
          \item \textbf{a meaningless labelling}, where every pixel borders another
                colour, so the ``boundary'' fills the plane.
        \end{enumerate}
        \smallskip
        Both give $D\approx2$ with a beautiful fit. \textbf{$D$ alone cannot tell
        them apart.} It detonates on slide 14.
      \end{block}
  \end{columns}
\end{frame}
\note{Plant the landmine and SAY you're planting it. This is the single most
important setup line in the talk. ~1 min.}

\begin{frame}{The Theorems We Lean On}
  \GeoPanels
  \vspace{0.15cm}
  \begin{block}{Uncertainty exponent \hfill {\normalfont Grebogi, McDonald, Ott \& Yorke (1983)}}
    Perturb a random initial condition by $\varepsilon$; let $f(\varepsilon)$ be the
    fraction whose final state changes. Then $f(\varepsilon)\sim\varepsilon^{\alpha}$ and
    \[ \boxed{\;D = d - \alpha\;}\qquad(d=2\ \text{for our slice}) \]
    \footnotesize
    $10\times$ more precision cuts your error by only $10^{\alpha}$. As $\alpha\to0$
    the boundary fills space and extra precision buys \emph{nothing}.
  \end{block}
  \vspace{0.05cm}
  \begin{block}{Basin entropy \hfill {\normalfont Daza, Wagemakers, Georgeot, Gu\'ery-Odelin \& Sanju\'an (2016)}}
    Cover with $\varepsilon$-boxes; in box $i$ with basin frequencies $p_{i,j}$, let
    $S_i=-\sum_j p_{i,j}\ln p_{i,j}$; average over \emph{boundary} boxes $\to S_{bb}$.
    \[ \boxed{\;S_{bb} > \ln 2 \;\Longrightarrow\; \text{fractal boundary}\;} \]
    \footnotesize A \emph{sufficient condition} --- a theorem, not a slope fit.
  \end{block}
\end{frame}
\note{Two results doing real work. GMOY: alpha near 0 => D near 2 => prediction
collapses. Daza: a theorem beats a line fit. Say that out loud. ~1.5 min.}

\begin{frame}{The System}
  \GeoPanels
  \vspace{0.15cm}
  \begin{columns}[T]
    \column{0.56\textwidth}
      Each of $N=83$ brain regions is a chaotic \textbf{Lorenz} oscillator,
      \[ \dot X=\sigma(Y-X),\quad \dot Y=X(\rho-Z)-Y,\quad \dot Z=XY-\beta Z \]
      with $\sigma=10,\ \rho=28,\ \beta=\tfrac83$. Regions are wired by a real human
      \textbf{DTI} scan; from its adjacency $A$ build $L=\mathrm{diag}(A\mathbf{1})-A$:
      \[ \boxed{\;\dot{\mathbf{x}}_i=f(\mathbf{x}_i)-K\textstyle\sum_j L_{ij}H\mathbf{x}_j\;} \]
      \footnotesize
      $H=e_1e_1^{\!\top}$ couples $x$-components; $K$ is the coupling --- the one knob.
    \column{0.40\textwidth}
      \centering
      \fig{dti_network.png}{83 regions, 850 edges, density 0.25, diameter 4.
      State space $83\times3=249$ dimensions.}
      \begin{block}{Why the Laplacian}
        \footnotesize
        $L\mathbf{1}=\mathbf{0}$, so full synchrony is \emph{always} an exact
        solution. The question is whether anything \emph{else} coexists.
      \end{block}
  \end{columns}
\end{frame}
\note{The Laplacian point earns 15 seconds: row sums zero => synchrony is always a
solution. So "does anything else survive?" is THE question. ~1.5 min.}

% ═════════════════════════════════════════════════════════ 9
\begin{frame}{My Project: Port the Paper's MATLAB to the GPU}
  \GeoPanels
  \vspace{0.2cm}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \begin{block}{The bottleneck}
        Resolving a fractal boundary needs a \emph{huge} grid of initial conditions
        --- each one an independent simulation. Embarrassingly parallel. The
        original MATLAB runs them one at a time.
      \end{block}
      \begin{block}{The expensive kernel: the VPS}
        \footnotesize
        The \textbf{Vector Pattern State} labels each run by \emph{how every pair of
        regions moves together}. For $N=83$ that is
        \[ \binom{83}{2}=3403 \text{ pairs},\qquad \text{VPS}\in\mathbb{R}^{6806} \]
        MATLAB computes it with a \texttt{for} loop over all 3403 pairs, calling
        \texttt{xcorr} on each.
      \end{block}
    \column{0.44\textwidth}
      \begin{block}{What the VPS actually is}
        \footnotesize
        For each pair $(i,j)$, from the cross-correlation:
        \begin{itemize}\setlength\itemsep{1pt}
          \item $\tau_{ij}$ = the \textbf{time lag} at the correlation peak
                --- \emph{how far out of phase} they are
          \item $L_{ij}$ = the norm of their difference \textbf{after aligning by
                that lag} --- how different once shifted
        \end{itemize}
        \smallskip
        The lag is the whole point: it detects \textbf{phase-lagged synchrony}, the
        signature of a chimera.
      \end{block}
  \end{columns}
\end{frame}
\note{This is the HPC heart. Say plainly: the science was blocked by compute, my job
was to unblock it. And define the VPS properly — tau is a TIME LAG. That matters in
three slides. ~1.5 min.}

% ═════════════════════════════════════════════════════════ 10
\begin{frame}[fragile]{The Port: 3403 Serial Correlations $\to$ One Batched FFT}
  \GeoPanels
  \vspace{0.15cm}
  \begin{columns}[T]
    \column{0.49\textwidth}
      \begin{block}{MATLAB}
        \begin{lstlisting}
for jk = 1:size(Vec,1)     % 3403 pairs, serial
  [cx lagsx] = xcorr(x(:,i), x(:,j));
  f = find(cx == max(cx));
  Tau_x(jk) = lagsx(f);
  ...
end
        \end{lstlisting}
      \end{block}
      \begin{block}{PyTorch, all pairs at once}
        \begin{lstlisting}
x_fft   = torch.fft.rfft(x, n=2T-1, dim=0)
corr_fft= x_fft[:,i] * torch.conj(x_fft[:,j])
corr    = torch.fft.irfft(corr_fft, n=2T-1)
tau     = torch.argmax(corr, dim=0)   # all 3403
        \end{lstlisting}
      \end{block}
    \column{0.47\textwidth}
      \begin{block}{The HPC content}
        \footnotesize
        \begin{itemize}\setlength\itemsep{2pt}
          \item \textbf{Wiener--Khinchin}: correlation becomes a product in Fourier
                space, $\mathcal{O}(T^2)\!\to\!\mathcal{O}(T\log T)$
          \item \textbf{Batching}: all 3403 pairs in one kernel --- the loop
                disappears into a tensor axis
          \item \textbf{Upper-triangular indexing}: only unique pairs stored
          \item \textbf{Batched RK4}: all initial conditions integrated
                simultaneously
          \item \textbf{SLURM job arrays}: one coupling $K$ per GPU
        \end{itemize}
      \end{block}
      \begin{block}{Scale reached on ACRES}
        \footnotesize
        $361^2 = 130{,}321$ initial conditions per coupling, 6806 features each,
        $\approx$1 GPU-hour per $K$, 20 couplings.
      \end{block}
  \end{columns}
  \vfill
  \centering{\footnotesize Benchmarked against a \textbf{serial CPU reference}
  (a line-by-line transcription of the MATLAB) --- same maths, same pair count.
  No MATLAB licence required, and it isolates the algorithm rather than the language.}
\end{frame}
\note{The engineering slide. The serial reference is the honest comparator: it's
the same algorithm, so we measure parallelisation, not MATLAB-vs-Python language
overhead. ~1.5 min.}


% ═════════════════════════════════════════════════════════ 10b
\begin{frame}[fragile]{The Port Was Slow --- and the FFT Wasn't Why}
  \GeoPanels
  \vspace{0.15cm}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \centering
      \fig{benchmark.png}{Left: runtime. Right: speedup over the serial
      reference. The looped port never clears $7\times$; vectorised reaches
      $375\times$.}
      \begin{block}{Profiling found it}
        \begin{lstlisting}
for lag_val in range(-T+1, T):   # 2T-1 = 19,999
    mask = tau == lag_val        #   iterations at T=1e4
        \end{lstlisting}
        \footnotesize
        The FFT parallelised $\tau$ --- then a \textbf{Python loop over every
        candidate lag} threw the win away.
      \end{block}
    \column{0.47\textwidth}
      \begin{block}{The fix: one gather, no loop}
        \footnotesize
        Each pair has its own lag, so build \textbf{per-pair aligned index grids}
        and gather once:
        \begin{lstlisting}
ti, tj = t + shift_i, t + shift_j
xa = x[ti, a_idx];  xb = x[tj, b_idx]
L  = norm((xa - xb) * valid, dim=0)
        \end{lstlisting}
      \end{block}
      \begin{block}{\alert{Result: $3\times \rightarrow 375\times$}}
        \footnotesize
        A further $\sim$57$\times$ over my own first port --- and the looped
        version was \textbf{slower than serial} at $T{=}128$ ($0.83\times$).
        Verified against the reference to float32 precision
        (max $|\Delta L| = 1.9\times10^{-6}$).
      \end{block}
  \end{columns}
\end{frame}
\note{THE HPC slide. The arc is the point: naive port 3x -> profile -> the
bottleneck is a Python loop, not the maths -> vectorise -> 375x, verified correct.
"The FFT was never the problem. The loop after it was." ~2 min.}

% ═════════════════════════════════════════════════════════ 11
\begin{frame}[fragile]{Did I Actually Reproduce It? {\normalfont(I tested. No.)}}
  \GeoPanels
  \vspace{0.15cm}
  A port is only a port if the numbers match. I transcribed
  \texttt{VectorPatternState.m} line-by-line into a reference implementation and
  compared, on the paper's own test matrices:
  \vspace{0.15cm}
  \begin{columns}[T]
    \column{0.46\textwidth}
      \centering
      \fig{alignment_bias.png}{Every pair lies above the diagonal.}
      {\scriptsize $\tau$ matches on \texttt{Example\_A\_3}, differs on
      \texttt{A\_1}/\texttt{A\_2} (tie-breaking). $L$ disagrees on
      \emph{every} input.}
    \column{0.50\textwidth}
      \begin{block}{The cause: an off-by-one --- in the \emph{original}}
        \begin{lstlisting}
% MATLAB, lag>0: aligns by  lag-1
norm( x(1:end-(lag-1), j) - x(lag:end, i) )
        \end{lstlisting}
        \vspace{-0.2cm}
        \begin{lstlisting}
# PyTorch, lag>0: aligns by  lag
x[lag:, i] - x[:-lag, j]
        \end{lstlisting}
        \footnotesize
        At $\text{lag}=1$ MATLAB's slices are \textbf{both the full series --- no shift
        at all}, despite a detected lag of~1.
      \end{block}
  \end{columns}
  \vfill
  {\footnotesize\color{Rust}\textbf{Measured:} the under-shift mis-aligns every
  pair, so it \textbf{inflates} $L$ --- $L_{\text{matlab}} \ge L_{\text{corrected}}$ on
  \textbf{36/36} pairs, mean ratio \textbf{1.96}. \textbf{So which is right?} To
  \emph{replicate} I must reproduce the original; to be \emph{correct} I must not.
  My code now ships \textbf{both}, so it is a measurement rather than an argument.}
\end{frame}
\note{This is the most grown-up slide in the deck. I tested my own port against the
source and it failed — and the discrepancy may be a bug in the published code. Do
NOT editorialise beyond "this needs a conversation." ~2 min.}

\begin{frame}{The Memory Wall --- and the Substitution It Forced}
  \GeoPanels
  \vspace{0.2cm}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \begin{block}{The real constraint}
        The true VPS needs the \textbf{whole trajectory} $(T\times N)$ per initial
        condition to cross-correlate.
        \smallskip
        At grid $64^2$, storing them is
        \[ \approx \textbf{41 GB} \]
        \smallskip
        So the production sweep uses a \textbf{streaming} approximation
        (Welford, single-pass, $\mathcal{O}(B\cdot C)$ memory) that never stores a
        trajectory.
      \end{block}
    \column{0.46\textwidth}
      \begin{block}{\alert{But the substitute isn't the VPS}}
        \footnotesize
        \begin{tabular}{ll}
          \toprule
          paper / port & streaming \\
          \midrule
          $\tau$ = time lag & $\tau$ = $\mathrm{mean}|\Delta X|/\mathrm{std}|\Delta X|$ \\
          $L$ after alignment & $L$ = instantaneous mean \\
          \bottomrule
        \end{tabular}
        \smallskip
        \textbf{No lag anywhere.} And the lag is what detects phase-lagged

\begin{frame}{CLV diagnostics: Ginelli's algorithm (implementation)}
  \GeoPanels
  \vspace{0.15cm}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \begin{block}{Algorithm overview}
        \footnotesize
        Ginelli's method recovers covariant Lyapunov vectors (CLVs) via a two-pass
        integration:
        \begin{enumerate}\setlength\itemsep{2pt}
          \item Forward pass: integrate the nonlinear state and a set of tangent
                vectors simultaneously (batched RK4). Every $q$ steps perform a QR
                factorisation $T=QR$ to re-orthonormalise the tangents.
          \item Store the upper-triangular $R_n$ factors (on-GPU, in \texttt{float16}
                to save VRAM) and the orthonormal $Q_n$ bases (offloaded to CPU).
          \item Backward pass: recurse $C_{n-1}=\text{solve}(R_n, C_n)$ with
                column-wise normalisation; the CLVs at time $n$ are $V_n=Q_n C_n$.
        \end{enumerate}
      \end{block}
      \begin{block}{Practical choices (this repo)}
        \footnotesize
        \begin{itemize}\setlength\itemsep{2pt}
          \item Batched RK4 kernel reuses the project's integrator and Jacobian API.
          \item QR performed every $\textbf{10}$ steps (configurable).
          \item R matrices stored as \texttt{torch.float16} on GPU; Q as \texttt{numpy.float32} on CPU.
          \item Backward linear solves cast R to \texttt{float32} for numerics then normalise columns.
        \end{itemize}
      \end{block}
    \column{0.43\textwidth}
      \centering
      \fig{clv_pipeline.png}{Forward QR every 10 steps; store $R_n$ in half precision; backward solve to reconstruct CLVs.}
  \end{columns}
  \vfill
  \footnotesize{Implementation: \texttt{pythongpu/pipeline/clv_diagnostics.py} provides \texttt{CLVCalculator} implementing the steps above and a diagnostic that computes pairwise transversality angles among the leading CLVs.}
\end{frame}

\begin{frame}{CLV transversality and riddled basins}
  \GeoPanels
  \vspace{0.15cm}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \begin{block}{Transversality angles}
        \footnotesize
        Given CLVs $v_i(t)$, compute pairwise angles $\theta_{ij}(t)=\arccos\big(|\langle v_i,v_j\rangle|\big)$.
        Small angles (near $0$) indicate near-tangencies in tangent space and are a
        precursor to basin riddling: loss of hyperbolicity and nearby trajectories
        becoming indistinguishable along unstable directions.
      \end{block}
      \begin{block}{Diagnostics produced}
        \footnotesize
        The pipeline computes the time-series of the minimum angle among the
        leading $K$ CLVs and writes it to \texttt{output/clv_angles_K\{K\}.npy}.
        Statistical summaries (fraction of times \(\min\theta<\epsilon\),\n median, and quantiles)
        are used to detect transitions into riddled regimes.
      \end{block}
    \column{0.43\textwidth}
      \centering
      \fig{clv_angles_example.png}{Example min-angle time series; sustained drops toward 0 indicate tangencies and potential riddling.}
    \end{columns}
\end{frame}
