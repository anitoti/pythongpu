% ═════════════════════════════════════════════════════════ 3  SYSTEM + THE PLANE
\begin{frame}{The System and the Viewing Plane}
  \GeoPanels
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.55\textwidth}
      Each of $N=83$ brain regions is a chaotic \textbf{Lorenz} oscillator; regions
      are wired by a real human \textbf{DTI} scan. With Laplacian
      $L=\mathrm{diag}(A\mathbf{1})-A$,
      \[ \boxed{\;\dot{\mathbf{x}}_i=f(\mathbf{x}_i)-K\displaystyle\sum_j L_{ij}H\mathbf{x}_j\;} \]
      \footnotesize
      $H=e_1e_1^{\!\top}$ couples $x$-components; $K$ is the coupling --- the one knob.
      \smallskip

      \normalsize
      The state space is $83\times3=\textbf{249}$ dimensions. You can't see a
      249-D basin, so \textbf{my project's task} is to pick a 2-D
      \textbf{plane of initial conditions}, sweep a grid, integrate each point, and
      colour by outcome. That coloured grid \emph{is} the basin map --- the ``view
      from a plane.''
    \column{0.42\textwidth}
      \centering
      \fig[0.42]{dti_adjacency.png}{83 regions, 850 edges, density 0.25, diameter 4.}
      \begin{block}{Why the Laplacian}
        \footnotesize
        $L\mathbf{1}=\mathbf{0}$, so full synchrony is \emph{always} an exact
        solution. The question is whether anything \emph{else} coexists ---
        i.e.\ whether the basin structure is rich.
      \end{block}
  \end{columns}
\end{frame}
\note{Ground the project scope: high-dim basin, choose a plane, colour it. The
Laplacian point earns 15 s — row sums zero => synchrony always solves it, so ``does
anything else survive?'' is THE question. ~1.5 min.}

% ═════════════════════════════════════════════════════════ 4  MEASURE + LANDMINE
\begin{frame}{Measuring Crinkliness: A Key Ambiguity}
  \GeoPanels
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{Box-counting dimension.} Cover the boundary with boxes of side
      $\varepsilon$; if $N(\varepsilon)\sim\varepsilon^{-D}$, then
      $D_0=\lim \log N(\varepsilon)/\log(1/\varepsilon)$.
      Smooth $\Rightarrow D=1$. Area-filling $\Rightarrow D=2$. Fractal in between.
      \medskip

      By Grebogi--McDonald--Ott--Yorke (1983), $D=d-\alpha$ ($d=2$ here). As
      $\alpha\to0$ the boundary fills space and \textbf{extra precision buys almost
      no prediction} --- a hard limit, not an engineering gap.
      \medskip

      {\footnotesize Fix the parameters and the system \emph{has} some definite
      number of attractors. We are \textbf{estimating an integer already out
      there}, not choosing it.}
    \column{0.44\textwidth}
      \begin{block}{Hold onto this --- it detonates on slide 9}
        $D\to2$ has \textbf{two} explanations:
        \smallskip
        \begin{enumerate}
          \item an intricately interwoven boundary --- the exciting one;
          \item \textbf{a meaningless labelling}, where every pixel borders another
                colour, so the ``boundary'' fills the plane.
        \end{enumerate}
        \smallskip
        Both give $D\approx2$ with a beautiful fit. \textbf{$D$ alone cannot tell
        them apart.}
      \end{block}
  \end{columns}
\end{frame}
\note{Plant the landmine and SAY you're planting it. Single most important setup
line in the talk. ~1.5 min.}

% ═════════════════════════════════════════════════════════ 5  MY PROJECT / THE PORT
\begin{frame}[fragile]{My Project: Porting the VPS to the GPU}
  \GeoPanels
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \begin{block}{The bottleneck}
        \footnotesize
        A fractal boundary needs a \emph{huge} grid of initial conditions --- each an
        independent simulation. Embarrassingly parallel. The original MATLAB runs
        them one at a time.
      \end{block}
      \begin{block}{The expensive kernel: the VPS}
        \footnotesize
        The \textbf{Vector Pattern State} labels each run by \emph{how every pair of
        regions moves together}: $\binom{83}{2}=3403$ pairs, VPS $\in\mathbb{R}^{6806}$.
        Per pair, from the cross-correlation:
        \begin{itemize}\setlength\itemsep{1pt}
          \item $\tau_{ij}$ = \textbf{time lag} at the peak (how out of phase);
          \item $L_{ij}$ = difference norm \emph{after aligning by that lag}.
        \end{itemize}
        The lag detects \textbf{phase-lagged synchrony} --- the chimera signature.
        MATLAB: a \texttt{for} loop over all 3403 pairs.
      \end{block}
    \column{0.47\textwidth}
      \begin{block}{The port: 3403 serial correlations $\to$ one batched FFT}
        \begin{lstlisting}
x_fft = torch.fft.rfft(x, n=2T-1, dim=0)
corr  = torch.fft.irfft(x_fft[:,i]*conj(x_fft[:,j]))
tau   = torch.argmax(corr, dim=0)   # all 3403
        \end{lstlisting}
        \footnotesize
        \begin{itemize}\setlength\itemsep{1pt}
          \item \textbf{Wiener--Khinchin}: $\mathcal{O}(T^2)\!\to\!\mathcal{O}(T\log T)$
          \item \textbf{Batching}: the loop becomes a tensor axis
          \item \textbf{Batched RK4}: all ICs on the plane at once
          \item \textbf{SLURM arrays}: one coupling $K$ per GPU
        \end{itemize}
        \smallskip
        Scale on ACRES: $\mathbf{361^2=130{,}321}$ ICs per coupling, $\approx$1
        GPU-hour per $K$.
      \end{block}
  \end{columns}
\end{frame}
\note{The HPC heart. The science was blocked by compute; my job was to unblock it.
Define the VPS properly — tau is a TIME LAG; that matters two slides on. ~1.5 min.}

% ═════════════════════════════════════════════════════════ 6  THE PORT WAS SLOW
\begin{frame}[fragile]{The Port Was Slow: Diagnosing the Bottleneck}
  \GeoPanels
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \centering
      \fig{benchmark.png}{Left: runtime. Right: speedup over the serial
      reference. Looped port never clears $7\times$; vectorised reaches $375\times$.}
      \begin{block}{Profiling found it}
        \begin{lstlisting}
for lag_val in range(-T+1, T):   # 2T-1 = 19,999
    mask = tau == lag_val        #   at T=1e4
        \end{lstlisting}
        \footnotesize
        The FFT parallelised $\tau$ --- then a \textbf{Python loop over every
        candidate lag} threw the win away.
      \end{block}
    \column{0.47\textwidth}
      \begin{block}{The fix: one gather, no loop}
        \footnotesize
        Each pair has its own lag, so build \textbf{per-pair aligned index grids}
        and gather once:
        \begin{lstlisting}
ti,tj = t+shift_i, t+shift_j
xa,xb = x[ti,a_idx], x[tj,b_idx]
L = norm((xa-xb)*valid, dim=0)
        \end{lstlisting}
      \end{block}
      \begin{block}{Result: $3\times \rightarrow 375\times$}
        \footnotesize
        $\sim$57$\times$ over my own first port. The looped version was
        \textbf{slower than serial} at $T{=}128$ ($0.83\times$). Verified against
        the reference to float32 (max $|\Delta L| = 1.9\times10^{-6}$).
      \end{block}
  \end{columns}
  \vfill
  \centering{\scriptsize Comparator is a \textbf{serial CPU reference} (a line-by-line
  MATLAB transcription) --- so this measures parallelisation, not language overhead.
  \pending{measured MATLAB-vs-GPU wall-clock (no MATLAB licence on our machines).}}
\end{frame}
\note{THE HPC slide. The arc IS the point: naive port 3x -> profile -> the
bottleneck is a Python loop, not the maths -> vectorise -> 375x, verified. ``The FFT
was never the problem. The loop after it was.'' ~2 min.}

% ═════════════════════════════════════════════════════════ 7  DID I REPRODUCE IT?
\begin{frame}[fragile]{Did I Actually Reproduce It?}
  \GeoPanels
  \vspace{0.1cm}
  A port is only a port if the numbers match. I transcribed
  \texttt{VectorPatternState.m} line-by-line and compared on the paper's own test
  matrices:
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.46\textwidth}
      \centering
      \fig[0.40]{alignment_bias.png}{Every pair lies above the diagonal.}
      {\scriptsize $\tau$ matches on some inputs, differs on others (tie-breaking);
      $L$ disagrees on \emph{every} input.}
    \column{0.50\textwidth}
      \begin{block}{The cause: an off-by-one --- in the \emph{original}}
        \begin{lstlisting}
% MATLAB, lag>0: aligns by  lag-1
norm( x(1:end-(lag-1), j) - x(lag:end, i) )
        \end{lstlisting}
        \vspace{-0.2cm}
        \begin{lstlisting}
# PyTorch, lag>0: aligns by  lag
x[lag:, i] - x[:-lag, j]
        \end{lstlisting}
        \footnotesize
        At $\text{lag}=1$ MATLAB's slices are \textbf{both the full series --- no
        shift at all}, despite a detected lag of 1.
      \end{block}
  \end{columns}
  \vfill
  {\footnotesize\color{Rust}\textbf{Measured:} the under-shift \textbf{inflates}
  $L$ --- $L_{\text{matlab}} \ge L_{\text{corrected}}$ on \textbf{36/36} pairs, mean
  ratio \textbf{1.96}. To \emph{replicate} I must reproduce the original; to be
  \emph{correct} I must not. My code ships \textbf{both} --- a measurement, not an
  argument. \textbf{This needs a conversation, not a unilateral fix.}}
\end{frame}
\note{The most grown-up slide. I tested my port against the source and it failed —
and the discrepancy may be a bug in the published code. Do NOT editorialise beyond
``this needs a conversation.'' ~1.5 min.}

% ═════════════════════════════════════════════════════════ 8  MEMORY WALL
\begin{frame}{The Memory Wall and Its Forced Substitution}
  \GeoPanels
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \begin{block}{The real constraint}
        The true VPS needs the \textbf{whole trajectory} $(T\times N)$ per initial
        condition to cross-correlate.
        \smallskip
        At grid $64^2$, storing them is $\approx \textbf{41 GB}$.
        \smallskip
        So the production sweep uses a \textbf{streaming} approximation
        (Welford, single-pass) that never stores a trajectory.
      \end{block}
    \column{0.46\textwidth}
      \begin{block}{But the substitute isn't the VPS}
        \footnotesize
        \begin{tabular}{@{}p{0.32\linewidth}p{0.55\linewidth}@{}}
          \toprule
          paper / port & streaming \\
          \midrule
          $\tau$ = time lag & $\tau$ = $\mathrm{mean}|\Delta X|/\mathrm{std}|\Delta X|$ \\
          $L$ after alignment & $L$ = instantaneous mean \\
          \bottomrule
        \end{tabular}
        \par\smallskip
        \textbf{No lag anywhere.} And the lag is what detects phase-lagged
        synchrony --- the chimera signature the VPS exists to find.
      \end{block}
  \end{columns}
  \vspace{0.15cm}
  \centering
  {\color{Rust}\textbf{Every result this summer came from the streaming path.}
  The memory wall silently changed the science.}
\end{frame}
\note{The real HPC lesson, and it's not ``I made it fast.'' A resource constraint
forced an approximation that quietly stopped computing the quantity the paper
defines. Nobody noticed for weeks — including me. ~1.5 min.}

% ═════════════════════════════════════════════════════════ 10  VALIDATE THE RULER
\begin{frame}{Validate the Ruler Before Trusting the Measurement}
  \GeoPanels
  \vspace{0.1cm}
  I needed an observable that needs \textbf{no clustering}: per-node long-time
  $\langle x\rangle$ (a time average, so constant on an attractor). Then check it on
  a system whose answer I \emph{already know}. At $K=0$ the network is uncoupled ---
  one ergodic attractor per node --- so long-time averages \emph{must} converge to a
  common value.
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \centering
      \fig{convergence_control.png}{$K=0$ tracks $1/\sqrt{T}$ as ergodicity
      demands; $K=0.5$ is flat over $32\times$ more averaging.}
    \column{0.44\textwidth}
      \begin{block}{Both innocent alternatives excluded}
        \footnotesize
        \begin{itemize}\setlength\itemsep{2pt}
          \item \textbf{Unconverged averages?} No --- $K=0$ converges under the
                identical metric.
          \item \textbf{Long transients?} No --- flat under \textbf{320$\times$}
                more burn-in (20 $\to$ 6400).
        \end{itemize}
        \smallskip
        Ergodicity forbids IC-dependent long-time averages on \emph{one} attractor.
        \textbf{So there are many.}
      \end{block}
  \end{columns}
\end{frame}
\note{The methodological spine: build the ruler, check it on a known answer, THEN
measure. Every conclusion I got without doing this was wrong. ~1.5 min.}

% ═════════════════════════════════════════════════════════ 11  THE MECHANISM
\begin{frame}{The Mechanism: Coupling Locks Nodes onto Lobes}
  \GeoPanels
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \begin{block}{The observable}
        \footnotesize
        A Lorenz attractor has two wings. Long-time $\langle X\rangle$ is
        \textbf{0} if a trajectory visits both, $\pm7.8$ if it stays
        \textbf{locked} on one.
      \end{block}
      \centering
      \fig{lobe_histogram.png}{Uncoupled: ergodic, $\langle X\rangle\to0$.
      Coupled: 97.8\% pinned to one wing at $\pm7$.}
    \column{0.46\textwidth}
      \begin{block}{So the basin label is exact}
        \footnotesize
        The network's attractor is a \textbf{pattern of $\pm$ lobes across 83 nodes}:
        \[ \boxed{\ \text{label} = \mathrm{sign}\langle X_i\rangle\ } \]
        83 bits. Discrete. \textbf{No clustering, no $k$, no elbow, no silhouette.}
        Up to $2^{83}\approx10^{25}$ candidate attractors.
      \end{block}
      \begin{block}{And they're permanent}
        \footnotesize
        Same ICs, two \textbf{disjoint} time windows: \textbf{99.9\%} of bits agree;
        94.6\% of ICs reproduce their exact 83-bit pattern.
      \end{block}
  \end{columns}
\end{frame}
\note{This is the finding. Uncoupled = ergodic = one attractor. Coupled = locked =
combinatorially many. The label is a FACT, not an estimate — that's what kills every
clustering problem in this talk. ~1.5 min.}

\begin{frame}{Where Does the Locking Switch On?}
  \GeoPanels
  \vspace{0.05cm}
  \begin{columns}[T]
    \column{0.56\textwidth}
      \centering
